-- ============================================================================ -- CSC Phase 2b — Storage-Bucket Setup -- -- Falls der Bucket "bewerbung-anhaenge" noch nicht existiert oder anders -- konfiguriert ist, hier das saubere Setup: -- - Privat (nicht oeffentlich lesbar) -- - 10 MB Datei-Limit -- - Erlaubte MIME-Types: pdf, jpeg, png, heic -- ============================================================================ -- 1. Bucket erstellen oder updaten INSERT INTO storage.buckets (id, name, public, file_size_limit, allowed_mime_types) VALUES ( 'bewerbung-anhaenge', 'bewerbung-anhaenge', false, 10485760, -- 10 MB in Bytes ARRAY['application/pdf', 'image/jpeg', 'image/jpg', 'image/png', 'image/heic'] ) ON CONFLICT (id) DO UPDATE SET public = false, file_size_limit = 10485760, allowed_mime_types = ARRAY['application/pdf', 'image/jpeg', 'image/jpg', 'image/png', 'image/heic']; -- 2. Storage-Policies (idempotent) -- INSERT: anon und authenticated duerfen hochladen DROP POLICY IF EXISTS "csc_bewerbung_anhaenge_upload" ON storage.objects; CREATE POLICY "csc_bewerbung_anhaenge_upload" ON storage.objects FOR INSERT TO anon, authenticated WITH CHECK (bucket_id = 'bewerbung-anhaenge'); -- SELECT: nur authenticated (Hauptverein-Admin, Vereinsvorstand) DROP POLICY IF EXISTS "csc_bewerbung_anhaenge_read_authenticated" ON storage.objects; CREATE POLICY "csc_bewerbung_anhaenge_read_authenticated" ON storage.objects FOR SELECT TO authenticated USING (bucket_id = 'bewerbung-anhaenge'); -- DELETE: nur authenticated DROP POLICY IF EXISTS "csc_bewerbung_anhaenge_delete_authenticated" ON storage.objects; CREATE POLICY "csc_bewerbung_anhaenge_delete_authenticated" ON storage.objects FOR DELETE TO authenticated USING (bucket_id = 'bewerbung-anhaenge'); -- 3. Bewerber-Anhaenge-Tabelle (falls nicht aus initialem Schema vorhanden) CREATE TABLE IF NOT EXISTS csc_bewerbung_anhaenge ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), bewerbung_id UUID REFERENCES csc_bewerbungen(id) ON DELETE CASCADE, storage_path TEXT NOT NULL, original_filename TEXT NOT NULL, mime_type TEXT NOT NULL, file_size INT NOT NULL, kategorie TEXT, -- 'lebenslauf', 'sachkunde', 'sonstiges' hochgeladen_am TIMESTAMPTZ DEFAULT now() ); -- RLS aktivieren ALTER TABLE csc_bewerbung_anhaenge ENABLE ROW LEVEL SECURITY; -- INSERT-Policy fuer anon DROP POLICY IF EXISTS "csc_anhaenge_insert_public" ON csc_bewerbung_anhaenge; CREATE POLICY "csc_anhaenge_insert_public" ON csc_bewerbung_anhaenge FOR INSERT TO anon, authenticated WITH CHECK (true); -- 4. Verifikation DO $$ DECLARE bucket_exists BOOLEAN; policy_count INT; BEGIN SELECT EXISTS (SELECT 1 FROM storage.buckets WHERE id = 'bewerbung-anhaenge') INTO bucket_exists; SELECT COUNT(*) INTO policy_count FROM pg_policies WHERE schemaname = 'storage' AND tablename = 'objects' AND policyname LIKE 'csc_bewerbung_anhaenge%'; IF bucket_exists THEN RAISE NOTICE 'OK: Bucket bewerbung-anhaenge existiert'; ELSE RAISE WARNING 'FEHLT: Bucket bewerbung-anhaenge nicht angelegt'; END IF; RAISE NOTICE 'Storage-Policies aktiv: %', policy_count; END $$; DO $$ BEGIN RAISE NOTICE '===================================='; RAISE NOTICE 'Phase 2b Storage-Setup angewendet'; RAISE NOTICE '===================================='; END $$;
