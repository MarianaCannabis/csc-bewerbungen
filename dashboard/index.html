-- ============================================================================ -- TRANCHE 2: SCORE-SYSTEM (korrigierte Version mit eindeutigen Dollar-Tags) -- ============================================================================ -- Hinweis: Jede Function nutzt einen eindeutigen Dollar-Tag ($body1$, $body2$), -- damit der Supabase SQL Editor die Function-Bodies sauber trennen kann. -- Bitte den GANZEN Block in einem Rutsch ausführen. -- ============================================================================ -- ---------------------------------------------------------------------------- -- 2.1) csc_compute_bewerbung_score(bewerbung_id) — Score-Berechnung -- ---------------------------------------------------------------------------- CREATE OR REPLACE FUNCTION csc_compute_bewerbung_score(p_bewerbung_id UUID) RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $body1$ DECLARE b RECORD; v_rolle TEXT; v_scores JSONB := '{}'::jsonb; v_gesamt INTEGER := 0; v_verfueg INTEGER := 0; v_erfahrung INTEGER := 0; v_mobilitaet INTEGER := 0; v_softskills INTEGER := 0; v_bereitschaft INTEGER := 0; v_sachkunde INTEGER := 0; v_gewicht JSONB; v_rolle_score INTEGER; v_slot_count INTEGER := 0; v_slot_total INTEGER := 0; v_tag_data JSONB; v_slot_val JSONB; BEGIN SELECT * INTO b FROM csc_bewerbungen WHERE id = p_bewerbung_id; IF NOT FOUND THEN RETURN NULL; END IF; -- Hard-Filter pruefen IF NOT COALESCE(b.ist_volljaehrig, false) THEN UPDATE csc_bewerbungen SET score_gesamt = 0, score_pro_rolle = jsonb_build_object('_hardfilter', 'nicht_volljaehrig') WHERE id = p_bewerbung_id; RETURN jsonb_build_object('score_gesamt', 0, 'hard_filter_fail', 'nicht_volljaehrig'); END IF; IF NOT COALESCE(b.arbeitserlaubnis, true) THEN UPDATE csc_bewerbungen SET score_gesamt = 0, score_pro_rolle = jsonb_build_object('_hardfilter', 'keine_arbeitserlaubnis') WHERE id = p_bewerbung_id; RETURN jsonb_build_object('score_gesamt', 0, 'hard_filter_fail', 'keine_arbeitserlaubnis'); END IF; -- FAKTOR 1: Verfuegbarkeit (0-100) - belegte Slots / max 35 v_slot_count := 0; v_slot_total := 0; IF b.verfuegbarkeit IS NOT NULL AND jsonb_typeof(b.verfuegbarkeit) = 'object' THEN FOR v_tag_data IN SELECT value FROM jsonb_each(b.verfuegbarkeit) LOOP IF jsonb_typeof(v_tag_data) = 'object' THEN FOR v_slot_val IN SELECT value FROM jsonb_each(v_tag_data) LOOP IF jsonb_typeof(v_slot_val) = 'array' THEN v_slot_total := v_slot_total + jsonb_array_length(v_slot_val); v_slot_count := v_slot_count + ( SELECT COUNT(*)::INTEGER FROM jsonb_array_elements(v_slot_val) elem WHERE elem::TEXT IN ('true', '1') OR elem = to_jsonb(true) ); ELSIF jsonb_typeof(v_slot_val) = 'boolean' THEN v_slot_total := v_slot_total + 1; IF v_slot_val = to_jsonb(true) THEN v_slot_count := v_slot_count + 1; END IF; END IF; END LOOP; END IF; END LOOP; END IF; IF v_slot_total > 0 THEN v_verfueg := LEAST(100, (v_slot_count * 100) / v_slot_total); ELSE v_verfueg := 0; END IF; -- FAKTOR 2: Erfahrung & Qualifikation (0-100) v_erfahrung := 30; IF length(COALESCE(b.vorerfahrung, '')) > 20 THEN v_erfahrung := v_erfahrung + 20; END IF; IF length(COALESCE(b.vorerfahrung, '')) > 100 THEN v_erfahrung := v_erfahrung + 15; END IF; IF COALESCE(b.bildungsabschluss, '') NOT IN ('', 'kein') THEN v_erfahrung := v_erfahrung + 15; END IF; IF b.sprachen IS NOT NULL AND jsonb_typeof(b.sprachen) = 'object' THEN v_erfahrung := v_erfahrung + LEAST(20, ( SELECT COUNT(*)::INTEGER * 5 FROM jsonb_each(b.sprachen) WHERE value IS NOT NULL AND value::TEXT NOT IN ('null', '""', '"null"') )); END IF; v_erfahrung := LEAST(v_erfahrung, 100); -- FAKTOR 3: Mobilitaet (0-100) v_mobilitaet := 50; IF b.max_anfahrt IS NOT NULL AND b.max_anfahrt::TEXT != '' THEN BEGIN v_mobilitaet := LEAST(100, 40 + b.max_anfahrt::INTEGER); EXCEPTION WHEN OTHERS THEN v_mobilitaet := 50; END; END IF; IF COALESCE(b.eigenes_fahrzeug, false) THEN v_mobilitaet := LEAST(100, v_mobilitaet + 10); END IF; -- FAKTOR 4: Soft Skills (0-100) - Mittelwert mal 20 v_softskills := 50; IF b.soft_skills IS NOT NULL AND jsonb_typeof(b.soft_skills) = 'object' THEN BEGIN v_softskills := COALESCE(( SELECT (AVG((value::TEXT)::INTEGER) * 20)::INTEGER FROM jsonb_each(b.soft_skills) WHERE value::TEXT ~ '^[0-9]+$' ), 50); EXCEPTION WHEN OTHERS THEN v_softskills := 50; END; END IF; v_softskills := LEAST(GREATEST(v_softskills, 0), 100); -- FAKTOR 5: Bereitschaft (0-100) v_bereitschaft := (CASE WHEN COALESCE(b.bereit_we, false) THEN 34 ELSE 0 END) + (CASE WHEN COALESCE(b.bereit_feiertag, false) THEN 33 ELSE 0 END) + (CASE WHEN COALESCE(b.bereit_nacht, false) THEN 33 ELSE 0 END); -- FAKTOR 6: Sachkunde (0-100) v_sachkunde := CASE WHEN b.praevention_sachkunde IS NOT NULL AND b.praevention_sachkunde NOT IN ('', 'nein', 'null') THEN 100 ELSE 0 END; -- Pro Rolle gewichteter Score IF b.rollen IS NULL OR array_length(b.rollen, 1) IS NULL THEN v_scores := '{}'::jsonb; ELSE FOREACH v_rolle IN ARRAY b.rollen LOOP v_gewicht := CASE LOWER(v_rolle) WHEN 'ausgabe' THEN '{"v":30,"e":20,"m":15,"s":20,"b":10,"k":5}'::jsonb WHEN 'grower' THEN '{"v":25,"e":35,"m":10,"s":10,"b":15,"k":5}'::jsonb WHEN 'head_grower' THEN '{"v":25,"e":40,"m":10,"s":10,"b":10,"k":5}'::jsonb WHEN 'vorstand' THEN '{"v":15,"e":40,"m":10,"s":25,"b":5,"k":5}'::jsonb WHEN 'stellv_vorstand' THEN '{"v":15,"e":40,"m":10,"s":25,"b":5,"k":5}'::jsonb WHEN 'praevention' THEN '{"v":15,"e":25,"m":10,"s":15,"b":5,"k":30}'::jsonb WHEN 'praeventionsbeauftragter' THEN '{"v":15,"e":25,"m":10,"s":15,"b":5,"k":30}'::jsonb WHEN 'sicherheit' THEN '{"v":30,"e":25,"m":15,"s":10,"b":15,"k":5}'::jsonb WHEN 'reinigung' THEN '{"v":40,"e":15,"m":20,"s":10,"b":10,"k":5}'::jsonb WHEN 'springer' THEN '{"v":45,"e":15,"m":15,"s":10,"b":10,"k":5}'::jsonb ELSE '{"v":25,"e":25,"m":15,"s":15,"b":10,"k":10}'::jsonb END; v_rolle_score := ( v_verfueg * (v_gewicht->>'v')::INTEGER + v_erfahrung * (v_gewicht->>'e')::INTEGER + v_mobilitaet * (v_gewicht->>'m')::INTEGER + v_softskills * (v_gewicht->>'s')::INTEGER + v_bereitschaft* (v_gewicht->>'b')::INTEGER + v_sachkunde * (v_gewicht->>'k')::INTEGER ) / 100; v_scores := v_scores || jsonb_build_object(v_rolle, v_rolle_score); END LOOP; END IF; -- Gesamt-Score = Mittelwert v_gesamt := COALESCE(( SELECT AVG((value::TEXT)::INTEGER)::INTEGER FROM jsonb_each(v_scores) WHERE value::TEXT ~ '^[0-9]+$' ), 0); -- DB updaten UPDATE csc_bewerbungen SET score_gesamt = v_gesamt, score_pro_rolle = v_scores WHERE id = p_bewerbung_id; RETURN jsonb_build_object( 'score_gesamt', v_gesamt, 'score_pro_rolle', v_scores, 'faktoren', jsonb_build_object( 'verfuegbarkeit', v_verfueg, 'erfahrung', v_erfahrung, 'mobilitaet', v_mobilitaet, 'softskills', v_softskills, 'bereitschaft', v_bereitschaft, 'sachkunde', v_sachkunde ) ); END; $body1$; GRANT EXECUTE ON FUNCTION csc_compute_bewerbung_score(UUID) TO authenticated; -- ---------------------------------------------------------------------------- -- 2.2) Trigger fuer automatische Score-Berechnung -- ---------------------------------------------------------------------------- CREATE OR REPLACE FUNCTION csc_trigger_compute_score() RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $body2$ BEGIN PERFORM csc_compute_bewerbung_score(NEW.id); RETURN NEW; END; $body2$; DROP TRIGGER IF EXISTS csc_compute_score_on_insert ON csc_bewerbungen; CREATE TRIGGER csc_compute_score_on_insert AFTER INSERT ON csc_bewerbungen FOR EACH ROW EXECUTE FUNCTION csc_trigger_compute_score(); DROP TRIGGER IF EXISTS csc_compute_score_on_update ON csc_bewerbungen; CREATE TRIGGER csc_compute_score_on_update AFTER UPDATE OF verfuegbarkeit, rollen, vorerfahrung, bildungsabschluss, sprachen, soft_skills, max_anfahrt, eigenes_fahrzeug, bereit_we, bereit_feiertag, bereit_nacht, praevention_sachkunde, ist_volljaehrig, arbeitserlaubnis ON csc_bewerbungen FOR EACH ROW EXECUTE FUNCTION csc_trigger_compute_score(); -- ---------------------------------------------------------------------------- -- 2.3) ONE-TIME: Scores fuer alle existierenden Bewerbungen berechnen -- ---------------------------------------------------------------------------- SELECT vorname || ' ' || nachname AS bewerber, status, csc_compute_bewerbung_score(id) AS scoring_result FROM csc_bewerbungen ORDER BY eingegangen_am DESC;
